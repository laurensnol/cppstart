find_package(Doxygen)
if(Doxygen_FOUND)
  set(DOXYGEN_EXTRACT_ALL YES)
  set(DOXYGEN_BUILTIN_STL_SUPPORT YES)

  doxygen_add_docs(docs cppstart
    WORKING_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}/include")
else()
  message(WARNING
    "Doxygen not found\n"
    "Documentation cannot be built")
endif()
